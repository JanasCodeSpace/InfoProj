54void CSegmentApproximator::douglasPeuckerRecursive(list<CInputPoint3D>& segment, std::list<CInputPoint3D>::iterator startItr, std::list<CInputPoint3D>::iterator endItr)
55{
56 if (segment.size() < 3) return; // min Groesse pro Seg 3
57 if (distance(startItr, endItr) == 2) return; // Zeigerabstand == 2
58 CInputPoint3D pStart; CInputPoint3D pEnd; // Variablen deklarieren
59
60
61 /* Startpunkt setzen */
62 pStart.setX(startItr->getX()); pStart.setY(startItr->getY()); pStart.setZ(startItr->getZ());
63 pStart.setTime(startItr->getTime());
64 pStart.setEulerMatrix(startItr->getEulerMatrix());
65
66 /* Endpunkt setzen */
67 pEnd.setX(endItr->getX()); pEnd.setY(endItr->getY()); pEnd.setZ(endItr->getZ());
68 pEnd.setTime(endItr->getTime());
69 pEnd.setEulerMatrix(endItr->getEulerMatrix());
70
71 double dist = 0.0, maxDist = 0.0; // dist und maxDist initialisieren
72 std::list<CInputPoint3D>::iterator maxItr, itr; // Zeiger bilden
73
74
75 /* am weitesten Entfernten Punkt suchen */
76 for (itr = startItr; itr != endItr; itr++)
77 {
78 CLine3D line = CLine3D(pStart, pEnd);
79 // calc distance
80 dist = itr->distanceTo(line);
81 if (dist > maxDist) {
82 maxDist = dist;
83 maxItr = itr;
84 }
85 }
86
87 if (maxDist <= maxDistance) {
88
89 segment.erase((++startItr), endItr); // Punkt loeschen
90 return;
91 }
92
93 /* Douglas Peucker erneut aufrufen */
94 douglasPeuckerRecursive(segment, startItr, maxItr);
95 douglasPeuckerRecursive(segment, maxItr, endItr);
96}

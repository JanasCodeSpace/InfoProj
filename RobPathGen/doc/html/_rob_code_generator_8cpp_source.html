76void CRobCodeGenerator::postProcessing(vector<CInputPoint3D>& path)
77{
78 COutputPoint3D p;
79 double timePrev = 1;
80
81 for (size_t s = 0; s < path.size(); s++) // Fuer jeden Punkt in dem Vector
82 {
83 p.set(path[s].getX(), path[s].getY(), path[s].getZ());
84 if (input.getSpeedManual())
85 {
86 if (input.getSpeed() > MAX_SPEED) //Wenn maximale Geschwindigkeit ueberschritten wird, Geschwindigkeit begrenzen
87 input.setSpeed(MAX_SPEED, true);
88 }
89 else
90 {
91 if (s == 0)
92 p.setSpeed(1); //Der erste Punkt(0) wird mit Standardgeschwindigkeit 1m/s angefahren.
93
94 else
95 p.setSpeed(calculateSpeed(path[s], s, timePrev)); //Die Geschwindigkeit zwischen den weiteren Punkten wird berechnet.
96 }
97
98 if (input.getOrientationManual()) // Wenn der Winkel vorgegeben ist diesen setzten
99 {
100 tuple <double, double, double> angles;
101 angles = input.getAngles();
102 p.setA(get<0>(angles));
103 p.setB(get<1>(angles));
104 p.setC(get<2>(angles));
105 }
106 else // Sonst den Winkel berechnen
107 calculateAngles(p, path[s]);
108 timePrev = path[s].getTime();
109 processedPath.push_back(p);
110 }
111
112}

114double CRobCodeGenerator::calculateSpeed(CInputPoint3D& p, size_t s, double timePrev)
115{
116 double distance = 0;
117 double time = 0;
118 double speed;
119
120 distance = processedPath[s - 1].distanceTo(p); //Strecke zwischen p und dem Punkt zuvor
121 time = p.getTime() - timePrev; //Zeit zwischen p-1 und p
122
123 speed = distance / time; // Berechnug Geschwindigkeit zwischen zwei Punkten
124
125 if (speed > MAX_SPEED) //Begrenzung auf maximale Geschwindigkeit, falls Trackerdaten hï¿½heren Wert aufweisen
126 speed = MAX_SPEED;
127
128 return speed; //Zuweisung der Geschwindigkeit
129}

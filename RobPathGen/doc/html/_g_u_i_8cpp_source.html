194void GUI::calculate()
195{
196
197 if (inputPathUI.isEmpty())
198 {
199 QMessageBox messageBox;
200 messageBox.critical(0, "Error", "Keine Datei ausgewaehlt!");
201 messageBox.setFixedSize(500, 200);
202 return;
203 }
204 if (outputPathUI.isEmpty())
205 {
206 QMessageBox messageBox;
207 messageBox.critical(0, "Error", "Kein Pfad ausgewaehlt!");
208 messageBox.setFixedSize(500, 200);
209 return;
210 }
211
212 try
213 {
214 string outputPath = outputPathUI.toUtf8().constData();
215 string inputPath = inputPathUI.toUtf8().constData();
216 ui.textBrowser->clear();
217
218 //logging Initialisieren
219 CLogging logging(outputPath, inputParameter.getLoggingManual());
220
221 //read Data
222 CInputParameter input;
223 input = inputParameter;
224 input.openFile(inputPath);
225 ui.textBrowser->insertPlainText("Datei eingelesen\n");
226
227 //moving Average
228
229 CMeanFilter meanFilter;
230 meanFilter.setWindowSize(meanLength);
231 meanFilter.mean(inputParameter.getPath(), logging);
232 ui.textBrowser->insertPlainText("Gleitender Mittelwert berechnet\n");
233
234 // Douglas-Peuker Algorithm
235
236 CSegmentApproximator segmentApproximator;
237 segmentApproximator.setmaxDistance(dpTolerance);
238 segmentApproximator.approx(meanFilter.getPath(), logging);
239 ui.textBrowser->insertPlainText("Douglas-Peuker-Algorithmus berechnet\n");
240
241 // Puts the Segments together to one path
242
243 CPathBuilder pathBuilder;
244 pathBuilder.createPath(segmentApproximator.getSegmentsApproxVector(), logging);
245 ui.textBrowser->insertPlainText("Pfad zusammengesetzt\n");
246
247 // Calculates Speed, Angle and generates the Output Data
248
249 CRobCodeGenerator codeGenerator(inputParameter);
250 codeGenerator.generateRobCode(pathBuilder.getPath(), outputPath, inputPath, logging);
251 ui.textBrowser->insertPlainText("Datei erstellt\n");;
252 }
253
254 catch (exception& e)
255 {
256 QMessageBox messageBox;
257 messageBox.critical(0, "Error", e.what());
258 messageBox.setFixedSize(500, 200);
259 return;
260 }
261}

55list<CInputPoint3D> CMeanFilter::calculateMean(list<CInputPoint3D>& segment)
56{
57 double sumX = 0, sumY = 0, sumZ = 0; // Variablen zum Speichern der Summe
58 double div = 0; // Variable zum Speichern des Teilers
59
60 CInputPoint3D p; //Point3D zum Zwischenspeichern
61
62 size_t inputSize = segment.size();
63
64 list<CInputPoint3D>::iterator it = segment.begin();
65 list<CInputPoint3D> newSegment;
66
67 for (size_t i = 0; i < inputSize - windowSize; ++i) //For each element in the Segment
68 {
69 sumX = 0, sumY = 0, sumZ = 0; // Variablen zum Speichern der Summe auf 0 zurueck setzen
70 div = 0; // Variable zum Speichern des Teilers auf 0 zurueck setzen
71 p.setTime(it->getTime());
72 p.setEulerMatrix(it->getEulerMatrix());
73 for (size_t j = i; j < i + windowSize; ++j) // Build the sums for the three points
74 {
75 sumX += it->getX();
76 sumY += it->getY();
77 sumZ += it->getZ();
78 div++;
79 it++;
80 }
81 for (size_t index = windowSize; index > 0; index--) // Pain, the iterator has to be set back
82 {
83 it--;
84 }
85 p.set(sumX / div, sumY / div, sumZ / div); // Calculate smoothed values
86 if(it != segment.end())
87 it++;
88 newSegment.push_back(p);
89 }
90 return newSegment;
91}

111void CInputParameter::openFile(string path)
112{
113 ifstream fin(path);
114 CInputPoint3D tmpPoint; // Zwischenspeicher zum konvertieren von tmpEuler in Point3D
115 CEulerMatrix tmpEuler; // Zwischenspeicher zum konverteiren von DummyMatrix in EulerMatrix
116 double x, y, z; // Punktkoordinaten
117 double x_prev = 0, y_prev = 0, z_prev = 0; // Zwischenspeicher fuer Punktkoordinaten
118 double timestamp; // Zeitstempel
119 int segmentCount = -1; // Segmentzaehler
120 float dummyMatrix[3][3]; // DummyMatrix zum speichern
121
122
123 if (!fin.is_open())
124 {
125 cerr << "Datei konnte nicht geoeffnet werden" << endl; // Fehler Datei konnte nicht geoeffnet werden.
126 }
127 string line;
128
129 while(getline(fin, line))
130 {
131 std::istringstream sStream (line);
132 sStream >> timestamp >> x >> y >> z >> dummyMatrix[0][0] >> dummyMatrix[0][1] >> dummyMatrix[0][2] // Zeile in die einzelnen Parameter zerlegen
133 >> dummyMatrix[1][0] >> dummyMatrix[1][1] >> dummyMatrix[1][2] >> dummyMatrix[2][0] >> dummyMatrix[2][1] >> dummyMatrix[2][2]; // und in DummyMatrix bzw. Variablen abspeichern
134
135 tmpEuler.setMatrix(dummyMatrix); // DummyMatrix[3][3] in EulerMatrix speichern
136 tmpPoint.setPoint(timestamp, x, y, z, tmpEuler.getEulerMatrix()); // Variablen und EulerWinkel in CPoint3D speichern
137
138 if (detectJump(tmpPoint, x_prev, y_prev, z_prev)) // if there is a jump in the data, start da new segment
139 {
140 segmentCount++; // neues Segement anlegen
141 initialPath.push_back(list<CInputPoint3D>()); // Punkt in Segent speichern
142 }
143
144 initialPath[segmentCount].push_back(tmpPoint); // Punkt in bestehendes Segment abspeichern
145
146 x_prev = x; // X-Wert zwischenspeichern
147 y_prev = y; // Y-Wert zwischenspeichern
148 z_prev = z; // Z-Wert zwischenspeichern
149 }
150 fin.close(); // Datei schliessen
151}

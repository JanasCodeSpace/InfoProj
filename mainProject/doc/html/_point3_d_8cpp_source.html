76double CPoint3D::distanceTo(CLine3D line)
77{
78 double bx, by, bz, rv_sq, dist, vp1, vp2, vp3; // Variablen Anlegen
79
80 /*
81 Vermessen wird der Punkt selbst
82
83 bx, by, bz == Vektordifferenz
84 rv_sq == Betrag des Linienvektors
85 dist == Distanz von Punkt zu Linie
86 vp1, vp2, vp3 == Vektorprodukte
87 */
88
89 int rvx = line.p1.x - line.p2.x; // Parameter X des Linienvektor berechnen
90 int rvy = line.p1.y - line.p2.y; // Parameter Y des Linienvektor berechnen
91 int rvz = line.p1.z - line.p2.z; // Parameter Z des Linienvektor berechnen
92
93 rv_sq = sqrt(((double)rvx * (double)rvx) + ((double)rvy * (double)rvy) + ((double)rvz * (double)rvz)); // Betrag des Linienvektor berechnen
94
95 bx = x - (double)line.p1.x; // X(Punkt) - X(Aufpunkt)
96 by = y - (double)line.p1.y; // Y(Punkt) - Y(Aufpunkt)
97 bz = z - (double)line.p1.z; // Z(Punkt) - Z(Aufpunkt)
98
99 vp1 = by * rvz - bz * rvy; // Parameter X Vektorprodukt
100 vp2 = bz * rvx - bx * rvz; // Parameter Y Vektorprodukt
101 vp3 = bx * rvy - by * rvx; // Parameter Z Vektorprodukt
102
103 dist = sqrt(vp1 * vp1 + vp2 * vp2 + vp3 * vp3) / rv_sq; // Betrag des Vektors berechnen
104
105 return dist;
106}

84void CInputParameter::openFile(string path)
85{
86 ifstream fin(path);
87 CInputPoint3D tmpPoint; // Zwischenspeicher zum konvertieren von tmpEuler in Point3D
88 CEulerMatrix tmpEuler; // Zwischenspeicher zum konverteiren von DummyMatrix in EulerMatrix
89 double x, y, z; // Punktkoordinaten
90 double x_prev = 0, y_prev = 0, z_prev = 0; // Zwischenspeicher für Punktkoordinaten
91 double timestamp; // Zeitstempel
92 int segmentCount = -1; // Segmentzähler
93 float dummyMatrix[3][3]; // DummyMatrix zum speichern
94
95
96 if (!fin.is_open())
97 {
98 cerr << "Datei konnte nicht geöffnet werden" << endl; // Fehler Datei konnte nicht geöffnet werden.
99 }
100 string line;
101
102 while(getline(fin, line))
103 {
104 std::istringstream sStream (line);
105 sStream >> timestamp >> x >> y >> z >> dummyMatrix[0][0] >> dummyMatrix[0][1] >> dummyMatrix[0][2] // Zeile in die einzelnen Parameter zerlegen
106 >> dummyMatrix[1][0] >> dummyMatrix[1][1] >> dummyMatrix[1][2] >> dummyMatrix[2][0] >> dummyMatrix[2][1] >> dummyMatrix[2][2]; // und in DummyMatrix bzw. Variablen abspeichern
107
108 tmpEuler.setMatrix(dummyMatrix); // DummyMatrix[3][3] in EulerMatrix speichern
109 tmpPoint.setPoint(timestamp, x, y, z, tmpEuler.getEulerMatrix()); // Variablen und EulerWinkel in CPoint3D speichern
110
111 if (detectJump(tmpPoint, x_prev, y_prev, z_prev)) // if there is a jump in the data, start da new segment
112 {
113 segmentCount++; // neues Segement anlegen
114 initialPath.push_back(list<CInputPoint3D>()); // Punkt in Segent speichern
115 }
116
117 initialPath[segmentCount].push_back(tmpPoint); // Punkt in bestehendes Segment abspeichern
118
119 x_prev = x; // X-Wert zwischenspeichern
120 y_prev = y; // Y-Wert zwischenspeichern
121 z_prev = z; // Z-Wert zwischenspeichern
122 }
123 fin.close(); // Datei schließen
124}

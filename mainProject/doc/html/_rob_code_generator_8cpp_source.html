113double CRobCodeGenerator::calculateSpeed(CInputPoint3D& p, size_t s, double timePrev)
114{
115 double distance = 0;
116 double time = 0;
117
118 distance = processedPath[s - 1].distanceTo(p); //Strecke zwischen p und dem Punkt zuvor
119 time = p.getTime() - timePrev; //Zeit zwischen p-1 und p
120
121 speed = distance / time; // Berechnug Geschwindigkeit zwischen zwei Punkten
122
123 if (speed > MAX_SPEED) //Begrenzung auf maximale Geschwindigkeit, falls Trackerdaten hï¿½heren Wert aufweisen

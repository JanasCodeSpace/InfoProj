108double CRobCodeGenerator::calculateSpeed(CInputPoint3D& p, size_t s, double timePrev)
109{
110 double distance = 0;
111 double time = 0;
112
113 distance = processedPath[s - 1].distanceTo(p); //Strecke zwischen p und dem Punkt zuvor
114 time = p.getTime() - timePrev; //Zeit zwischen p-1 und p
115
116 speed = distance / time; // Berechnug Geschwindigkeit zwischen zwei Punkten
117

81double CSegmentApproximator::calcDist(int xS, int yS, int zS, int xE, int yE, int zE, int x, int y, int z)
82{
83 /*
84 xS, yS, zS == Startpunkt
85 xE, yE, zE == Endpunkt
86 x, y, z == Vermessungspunkt
87 */
88
89 //Crossproduct
90 double bx, by, bz, rv_sq, dist, vp1, vp2, vp3;
91
92 /*
93 bx, by, bz == Vektordifferenz
94 rv_sq == Betrag des Linienvektors
95 dist == Distanz von Punkt zu Linie
96 vp1, vp2, vp3 == Vektorprodukte
97 */
98
99 int rvx = xS - xE; // Parameter X des Linienvektor berechnen
100 int rvy = yS - yE; // Parameter Y des Linienvektor berechnen
101 int rvz = zS - zE; // Parameter Z des Linienvektor berechnen
102
103 rv_sq = sqrt((rvx * rvx) + (rvy * rvy) + (rvz * rvz)); // Betrag des Linienvektor berechnen
104
105 bx = x - xS; // X(Punkt) - X(Aufpunkt)
106 by = y - yS; // Y(Punkt) - Y(Aufpunkt)
107 bz = z - zS; // Z(Punkt) - Z(Aufpunkt)
108
109 vp1 = by * rvz - bz * rvy; // Parameter X Vektorprodukt
110 vp2 = bz * rvx - bx * rvz; // Parameter Y Vektorprodukt
111 vp3 = bx * rvy - by * rvx; // Parameter Z Vektorprodukt
112
113 dist = sqrt(vp1 * vp1 + vp2 * vp2 + vp3 * vp3) / rv_sq; // Betrag des Vektors berechnen
114
115 return dist;
116
117}

49void CSegmentApproximator::douglasPeuckerRecursive(list<CInputPoint3D>& segment, std::list<CInputPoint3D>::iterator startItr, std::list<CInputPoint3D>::iterator endItr)
50{
51 if (segment.size() < 3) return;
52 if (distance(startItr, endItr) == 2) return;
53 CInputPoint3D pStart; CInputPoint3D pEnd;
54 pStart.setX(startItr->getX()); pStart.setY(startItr->getY()); pStart.setZ(startItr->getZ());
55
56 pEnd.setX(endItr->getX()); pEnd.setY(endItr->getY()); pEnd.setZ(endItr->getZ());
57
58 double dist = 0.0, maxDist = 0.0;
59 std::list<CInputPoint3D>::iterator maxItr, itr;
60
61 for (itr = startItr; itr != endItr; itr++)
62 {
63 CLine3D line = CLine3D(pStart, pEnd);
64 // calc distance
65 dist = itr->distanceTo(line);
66 if (dist > maxDist) {
67 maxDist = dist;
68 maxItr = itr;
69 }
70 }
71
72 if (maxDist <= maxDistance) {
73
74 segment.erase((++startItr), endItr);
75 return;
76 }
77
78 douglasPeuckerRecursive(segment, startItr, maxItr);
79 douglasPeuckerRecursive(segment, maxItr, endItr);
80}
